鬠ｿ'use client'

import { useEffect, useState } from 'react'
import { createClient } from '@supabase/supabase-js'
import RecurringPayoutButton from '@/components/RecurringPayoutButton'

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL,
  process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY
)

const TYPE_LABEL = {
  one_shot: 'One shot',
  affiliation: 'Affiliation',
  ambassador: 'Ambassadeur',
  placement_produit: 'Placement de produit',
  ugc: 'UGC',
  notoriete: 'Notori笏慊ｮt笏慊ｮ',
}

const FREQ_LABEL = {
  weekly: 'Hebdomadaire',
  biweekly: 'Toutes les 2 semaines',
  monthly: 'Mensuel',
}

const fmtEUR = (n) =>
  `${(Number(n) || 0).toLocaleString('fr-FR', { minimumFractionDigits: 0, maximumFractionDigits: 2 })} ﾃ氾ｩﾂｼ`

const formatDate = (d) => (d ? new Date(d).toLocaleDateString('fr-FR') : 'ﾃ氾僌ｶ')

export default function ContratsSection({ user: userProp }) {
  const [loading, setLoading] = useState(true)
  const [contracts, setContracts] = useState([])
  const [error, setError] = useState(null)

  useEffect(() => {
    let cancelled = false

    async function load() {
      try {
        // 1. R笏慊ｮcup笏慊ｮrer l'user (prop ou session)
        let userId = userProp?.id
        if (!userId) {
          const { data } = await supabase.auth.getUser()
          userId = data?.user?.id
        }
        if (!userId) {
          if (!cancelled) { setError('Utilisateur non connect笏慊ｮ.'); setLoading(false) }
          return
        }

        // 2. Marque li笏慊ｮe 笏愿｡ l'user
        const { data: brand } = await supabase
          .from('brands')
          .select('id')
          .eq('user_id', userId)
          .single()
        if (!brand) {
          if (!cancelled) { setError('Profil marque introuvable.'); setLoading(false) }
          return
        }

        // 3. Contrats de la marque
        const { data: rows, error: contractsErr } = await supabase
          .from('contracts')
          .select('*, influencers(display_name), collaborations(campaigns(title))')
          .eq('brand_id', brand.id)
          .order('created_at', { ascending: false })

        if (contractsErr) {
          if (!cancelled) { setError(contractsErr.message); setLoading(false) }
          return
        }

        if (!cancelled) {
          setContracts(rows || [])
          setLoading(false)
        }
      } catch (e) {
        if (!cancelled) { setError(e.message || 'Erreur de chargement.'); setLoading(false) }
      }
    }

    load()
    return () => { cancelled = true }
  }, [userProp])

  if (loading) {
    return <div className="p-8 text-center text-gray-400">Chargement des contratsﾃ氾傔ｪ</div>
  }

  if (error) {
    return <div className="p-8 text-center text-red-600">{error}</div>
  }

  return (
    <div className="mx-auto max-w-3xl p-6">
      <div className="mb-6">
        <h2 className="text-2xl font-bold text-gray-900">Mes contrats</h2>
        <p className="mt-1 text-sm text-gray-500">
          G笏慊ｮrez vos contrats et activez les versements r笏慊ｮcurrents.
        </p>
      </div>

      {contracts.length === 0 ? (
        <div className="rounded-2xl border border-dashed border-gray-200 bg-white p-10 text-center text-gray-500">
          Aucun contrat pour le moment.
        </div>
      ) : (
        <div className="space-y-4">
          {contracts.map((c) => {
            const isRecurring = !!c.payout_frequency
            const hasSubscription = !!c.stripe_subscription_id
            const campaignTitle = c.collaborations?.campaigns?.title
            const influencerName = c.influencers?.display_name

            return (
              <div key={c.id} className="rounded-2xl border border-gray-200 bg-white p-5 shadow-sm">
                <div className="flex items-start justify-between gap-4">
                  <div>
                    <p className="font-semibold text-gray-900">
                      {campaignTitle || 'Contrat'}
                    </p>
                    <p className="text-sm text-gray-500">
                      {influencerName ? `Avec @${influencerName}` : 'Cr笏慊ｮateur'}
                    </p>
                  </div>
                  <span className="rounded-full bg-[#F3EFFF] px-3 py-1 text-xs font-semibold text-[#7C3AED]">
                    {TYPE_LABEL[c.contract_type] || c.contract_type || 'Contrat'}
                  </span>
                </div>

                <div className="mt-4 grid grid-cols-2 gap-3 text-sm sm:grid-cols-3">
                  <div>
                    <p className="text-gray-400">Montant</p>
                    <p className="font-medium text-gray-900">{fmtEUR(c.amount)}</p>
                  </div>
                  <div>
                    <p className="text-gray-400">Paiement</p>
                    <p className="font-medium text-gray-900">
                      {isRecurring ? (FREQ_LABEL[c.payout_frequency] || c.payout_frequency) : 'Unique'}
                    </p>
                  </div>
                  <div>
                    <p className="text-gray-400">Deadline</p>
                    <p className="font-medium text-gray-900">{formatDate(c.deadline)}</p>
                  </div>
                </div>

                {/* Zone versements r笏慊ｮcurrents */}
                {isRecurring && (
                  <div className="mt-4 border-t border-gray-100 pt-4">
                    {hasSubscription ? (
                      <span className="inline-flex items-center gap-2 rounded-lg bg-green-50 px-3 py-2 text-sm font-medium text-green-700">
                        ﾃ板｣ﾃｴ Versements r笏慊ｮcurrents actifs ({FREQ_LABEL[c.payout_frequency] || c.payout_frequency})
                      </span>
                    ) : (
                      <div className="flex flex-col gap-2">
                        <p className="text-sm text-gray-500">
                          Versement de {fmtEUR(c.payout_amount)} {(FREQ_LABEL[c.payout_frequency] || c.payout_frequency).toLowerCase()} au cr笏慊ｮateur.
                        </p>
                        <RecurringPayoutButton contractId={c.id} />
                      </div>
                    )}
                  </div>
                )}

                {/* Lien PDF si dispo */}
                {c.pdf_url && (
                  <div className="mt-3">
                    <a
                      href={c.pdf_url}
                      target="_blank"
                      rel="noopener noreferrer"
                      className="text-sm font-medium text-[#7C3AED] hover:underline"
                    >
                      Voir le contrat (PDF)
                    </a>
                  </div>
                )}
              </div>
            )
          })}
        </div>
      )}
    </div>
  )
}
